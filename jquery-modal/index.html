jQuery - Modal

A full responsive, easy-to-use modal plugin based on jQuery. Optimized for mobile browsers, including Apple Safari.

  • Source

Requirements

  • jQuery
  • modal.js
  • modal.css

add this to your head 

<link rel="stylesheet" href="modal.css">

add this to your head or before body end 

<script src="jquery.js"></script>
<script src="modal.js"></script>

Settings

content

content to show initial

  • type: string
  • default: ''
$.modal({
    content: '<h1>Hello World</h1>'
});

contentUrl

content from an external url (ajax-loaded)

  • type: string
  • default: null
$.modal({
    contentUrl: 'https://www.yourhomepage.org/content'
});

useCache

don't repeat ajax-load every time

  • type: boolean
  • default: true
$.modal({
    contentUrl: 'https://www.yourhomepage.org/content',
    useCache: false // load content every time
});

youtubeId

show youtube-iframe

  • type: string
  • default: null
$.modal({
    youtubeId: 'MNyG-xu-7SQ'
});

class

custom class for modal

  • type: string
  • default: ''
$.modal({
    class: '.mymodal'
});

closeBtn

show x-close-btn

  • type: boolean
  • default: true
$.modal({
    closeBtn: false
});

layerClose

modal closes when layer is clicked

  • type: boolean
  • default: true
$.modal({
    layerClose: false
});

closingSelectors

custom selectors for elements to close modal

  • type: array
  • default: null
$.modal({
    closingSelectors:  [
        '#mymodal-close-btn',
        '.modal-close'
    ]
});

showOnInit

show modal when created

  • type: boolean
  • default: true
$.modal({
    showOnInit: false
});

fadeInDuration

how fast modal fades in (milliseconds)

  • type: integer
  • default: 400
$.modal({
    fadeInDuration: 0 // apears immediately
});

fadeOutDuration

how fast modal fades out (milliseconds)

  • type: integer
  • default: 400
$.modal({
    fadeOutDuration: 0 // disapears immediately
});

beforeModalOpen

callback before modal appears

  • type: function
  • default: null
  • returns: object
$.modal({
    beforeModalOpen: (modal)=> console.log('beforeOpen', modal)
});

afterModalOpen

callback after modal appears

  • type: function
  • default: null
  • returns: object
$.modal({
    afterModalOpen: (modal)=> console.log('afterOpen', modal)
});

beforeModalClose

callback before modal disappears

  • type: function
  • default: null
  • returns: object
$.modal({
    beforeModalClose: (modal)=> console.log('beforeClose', modal)
});

afterModalClose

callback after modal disappears

  • type: function
  • default: null
  • returns: object
$.modal({
    afterModalClose: (modal)=> console.log('afterClose', modal)
});
