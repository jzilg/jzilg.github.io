Modal with static content

$('#demo-btn-default').on('click', function() {
    $.modal({
        content: '<div style="padding: 20px;"><h1>Hallo Welt</h1><p> Stet clita kasd gubergren, no sea takimata sanctus est Lorem ipsum dolor sit amet. Lorem ipsum dolor sit amet, consetetur sadipscing elitr, sed diam nonumy eirmod tempor invidunt ut labore et dolore magna aliquyam erat, sed diam voluptua. At vero eos et accusam et justo duo dolores et ea rebum.<hr /><button class="modal-close">Close</button><button class="modal-close2">Do not close</button></p></div>',
        class: 'my-modal',
        closingSelectors: [
            '.modal-close'
        ],
        fadeInDuration: 600,
        fadeOutDuration: 100,
        beforeModalOpen: (modal)=> console.log('beforeOpen', modal),
        afterModalOpen: (modal)=> console.log('afterOpen', modal),
        beforeModalClose: (modal)=> console.log('beforeClose', modal),
        afterModalClose: (modal)=> console.log('afterClose', modal)
    });
});

Api

content

content to show initial

  • type: string
  • default: ''
$.modal({
    content: '<h1>Hello World</h1>'
});

contentUrl

content from an external url (ajax-loaded)

  • type: string
  • default: null
$.modal({
    contentUrl: 'https://www.yourhomepage.org/content'
});

useCache

don't repeat ajax-load every time

  • type: boolean
  • default: true
$.modal({
    contentUrl: 'https://www.yourhomepage.org/content',
    useCache: false // load content every time
});

youtubeId

show youtube-iframe

  • type: string
  • default: null
$.modal({
    youtubeId: 'MNyG-xu-7SQ'
});

class

custom class for modal

  • type: string
  • default: ''
$.modal({
    class: '.mypopup'
});

closeBtn

show x-close-btn

  • type: boolean
  • default: true
$.modal({
    closeBtn: false
});

layerClose

modal closes when layer is clicked

  • type: boolean
  • default: true
$.modal({
    layerClose: false
});

closingSelectors

custom selectors for elements to close modal

  • type: array
  • default: null
$.modal({
    closingSelectors:  [
        '#mypopup-close-btn',
        '.modal-close'
    ]
});

showOnInit

show modal when created

  • type: boolean
  • default: true
$.modal({
    showOnInit: false
});

fadeInDuration

how fast modal fades in (milliseconds)

  • type: integer
  • default: 400
$.modal({
    fadeInDuration: 0 // apears immediately
});

fadeOutDuration

how fast modal fades out (milliseconds)

  • type: integer
  • default: 400
$.modal({
    fadeOutDuration: 0 // disapears immediately
});

beforeModalOpen

callback before modal appears

  • type: function
  • default: null
  • returns: object
$.modal({
    beforeModalOpen: (modal)=> console.log('beforeOpen', modal)
});

afterModalOpen

callback after modal appears

  • type: function
  • default: null
  • returns: object
$.modal({
    afterModalOpen: (modal)=> console.log('afterOpen', modal)
});

beforeModalClose

callback before modal disappears

  • type: function
  • default: null
  • returns: object
$.modal({
    beforeModalClose: (modal)=> console.log('beforeClose', modal)
});

afterModalClose

callback after modal disappears

  • type: function
  • default: null
  • returns: object
$.modal({
    afterModalClose: (modal)=> console.log('afterClose', modal)
});
